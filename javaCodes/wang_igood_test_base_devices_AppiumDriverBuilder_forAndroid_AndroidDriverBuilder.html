public class AndroidDriverBuilder implements Serializable{

    private Object endpoint; //
    private String xcodeOrgId; //
    private Object capabilities; //

    public Object getEndpoint(){
        return endpoint;
    }

    public void setEndpoint(Object endpoint){
        this.endpoint = endpoint;
    }

    public String getXcodeOrgId(){
        return xcodeOrgId;
    }

    public void setXcodeOrgId(String xcodeOrgId){
        this.xcodeOrgId = xcodeOrgId;
    }

    public Object getCapabilities(){
        return capabilities;
    }

    public void setCapabilities(Object capabilities){
        this.capabilities = capabilities;
    }

}
